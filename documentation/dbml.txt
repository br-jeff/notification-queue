Table plans {
    id uuid [primary key]
    name string
    messages_per_month integer
    created_at timestamp 
  }
  
  Table companies {
    id uuid [primary key]
    plan_id varchar
    created_at timestamp
  }
  
  
  Table users {
    id uuid [primary key]
    company_id uuid
    name varchar
    created_at timestamp
  }
  
  Table count_messages {
    id string //
    company_id uuid [primary key]
    date date
  }
  
  Table messages {
    id string [primary key]
    content text
    company_id uuid 
    created_at timestamp 
  }
  
  Table recipients {
    id uuid [primary key]
    messages_id text
    user_id uuid 
    is_open boolean
    created_at timestamp
  }
  
  Ref: plans.id > companies.plan_id 
  Ref: users.company_id  >  companies.id 
  Ref: companies.id > count_messages.company_id 
  Ref: companies.id > messages.company_id 
  Ref: recipients.messages_id > messages.id 
  Ref: recipients.user_id > users.id
